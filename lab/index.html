how it works

/* 1. 사선 패턴을 배경으로 만든다 */
.hatch-text {
  background-image: repeating-linear-gradient(
    -45deg,
    transparent          /* 투명 */   0px,
    transparent                          5px,
    rgba(255,255,255,0.35) /* 흰 선 */ 5px,
    rgba(255,255,255,0.35)               8px
  );
  background-color: #c05540; /* 베이스 색 */

  /* 2. 배경을 텍스트 모양으로 클리핑한다 */
  -webkit-background-clip: text;
  background-clip: text;
  color: transparent; /* 텍스트를 투명하게 → 배경이 보임 */

  /* 3. 배경 위치를 움직여 사선이 흐르는 효과 */
  animation: hatch-slide 1.2s linear infinite;
}

@keyframes hatch-slide {
  from { background-position: 0 0; }
  to   { background-position: 40px 0; }
}
